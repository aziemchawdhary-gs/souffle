119 void TypeDeclarationChecker::checkUnionType(const ast::UnionType& type) {
120  // check presence of all the element types and that all element types are based off a primitive
121  for (const QualifiedName& sub : type.getTypes()) {
122  if (typeEnv.isPrimitiveType(sub)) {
123  continue;
124  }
125  const ast::Type* subtype = getIf(
126  program.getTypes(), [&](const ast::Type* type) { return type->getQualifiedName() == sub; });
127  if (subtype == nullptr) {
128  report.addError(tfm::format("Undefined type %s in definition of union type %s", sub,
129  type.getQualifiedName()),
130  type.getSrcLoc());
131  } else if (!isA<ast::UnionType>(subtype) && !isA<ast::SubsetType>(subtype)) {
132  report.addError(tfm::format("Union type %s contains the non-primitive type %s",
133  type.getQualifiedName(), sub),
134  type.getSrcLoc());
135  }
136  }
137 
138  // Check if the union is recursive.
139  if (typeEnvAnalysis.isCyclic(type.getQualifiedName())) {
140  report.addError("Infinite descent in the definition of type " + toString(type.getQualifiedName()),
141  type.getSrcLoc());
142  }
143 
144  /* check that union types do not mix different primitive types */
145  for (const auto* type : program.getTypes()) {
146  // We are only interested in unions here.
147  if (!isA<ast::UnionType>(type)) {
148  continue;
149  }
150 
151  const auto& name = type->getQualifiedName();
152 
153  const auto& predefinedTypesInUnion = typeEnvAnalysis.getPrimitiveTypesInUnion(name);
154 
155  // Report error (if size == 0, then the union is cyclic)
156  if (predefinedTypesInUnion.size() > 1) {
157  report.addError(
158  tfm::format("Union type %s is defined over {%s} (multiple primitive types in union)",
159  name, join(predefinedTypesInUnion, ", ")),
160  type->getSrcLoc());
161  }
162  }
163 }

165 void TypeDeclarationChecker::checkRecordType(const ast::RecordType& type) {
166  auto&& fields = type.getFields();
167  // check proper definition of all field types
168  for (auto&& field : fields) {
169  if (!typeEnv.isType(field->getTypeName())) {
170  report.addError(tfm::format("Undefined type %s in definition of field %s", field->getTypeName(),
171  field->getName()),
172  field->getSrcLoc());
173  }
174  }
175 
176  // check that field names are unique
177  for (std::size_t i = 0; i < fields.size(); i++) {
178  auto&& cur_name = fields[i]->getName();
179  for (std::size_t j = 0; j < i; j++) {
180  if (fields[j]->getName() == cur_name) {
181  report.addError(tfm::format("Doubly defined field name %s in definition of type %s", cur_name,
182  type.getQualifiedName()),
183  fields[i]->getSrcLoc());
184  }
185  }
186  }

202 void TypeDeclarationChecker::checkSubsetType(const ast::SubsetType& astType) {
203  if (typeEnvAnalysis.isCyclic(astType.getQualifiedName())) {
204  report.addError(
205  tfm::format("Infinite descent in the definition of type %s", astType.getQualifiedName()),
206  astType.getSrcLoc());
207  return;
208  }
209 
210  if (!typeEnv.isType(astType.getBaseType())) {
211  report.addError(tfm::format("Undefined base type %s in definition of type %s", astType.getBaseType(),
212  astType.getQualifiedName()),
213  astType.getSrcLoc());
214  return;
215  }
216 
217  auto& rootType = typeEnv.getType(astType.getBaseType());
218 
219  if (isA<analysis::UnionType>(rootType)) {
220  report.addError(tfm::format("Subset type %s can't be derived from union %s",
221  astType.getQualifiedName(), rootType.getName()),
222  astType.getSrcLoc());
223  }
224 
225  if (isA<analysis::RecordType>(rootType)) {
226  report.addError(tfm::format("Subset type %s can't be derived from record type %s",
227  astType.getQualifiedName(), rootType.getName()),
228  astType.getSrcLoc());
229  }

232 void TypeDeclarationChecker::run() {
233  // The redefinitions of types is checked by checkNamespaces in SemanticChecker
234  for (auto* type : program.getTypes()) {
235  if (typeEnv.isPrimitiveType(type->getQualifiedName())) {
236  report.addError("Redefinition of the predefined type", type->getSrcLoc());
237  continue;
238  }
239 
240  if (isA<ast::UnionType>(type)) {
241  checkUnionType(*as<ast::UnionType>(type));
242  } else if (isA<ast::RecordType>(type)) {
243  checkRecordType(*as<ast::RecordType>(type));
244  } else if (isA<ast::SubsetType>(type)) {
245  checkSubsetType(*as<ast::SubsetType>(type));
246  } else if (isA<ast::AlgebraicDataType>(type)) {
247  checkADT(*as<ast::AlgebraicDataType>(type));
248  } else {
249  fatal("unsupported type construct: %s", typeid(type).name());
250  }
251  }
252 
253  // Check if all the branch names are unique in sum types.
254  std::map<std::string, std::vector<SrcLocation>> branchToLocation;
255  visitDepthFirst(program.getTypes(), [&](const ast::AlgebraicDataType& type) {
256  for (auto* branch : type.getBranches()) {
257  branchToLocation[branch->getConstructor()].push_back(branch->getSrcLoc());
258  }
259  });
260 
261  for (auto& branchLocs : branchToLocation) {
262  auto& branch = branchLocs.first;
263  auto& locs = branchLocs.second;
264 
265  // If a branch is used only once, then everything is fine.
266  if (locs.size() == 1) continue;
267 
268  auto primaryDiagnostic =
269  DiagnosticMessage(tfm::format("Branch %s is defined multiple times", branch));
270 
271  std::vector<DiagnosticMessage> branchDeclarations;
272  for (auto& loc : locs) {
273  branchDeclarations.push_back(DiagnosticMessage(tfm::format("Branch %s defined", branch), loc));
274  }
275 
276  report.addDiagnostic(Diagnostic(
277  Diagnostic::Type::ERROR, std::move(primaryDiagnostic), std::move(branchDeclarations)));
278  }
279 }

281 void TypeCheckerImpl::visitAtom(const Atom& atom) {
282  auto relation = getAtomRelation(&atom, &program);
283  if (relation == nullptr) {
284  return; // error unrelated to types.
285  }
286 
287  auto attributes = relation->getAttributes();
288  auto arguments = atom.getArguments();
289  if (attributes.size() != arguments.size()) {
290  return; // error in input program
291  }
292 
293  for (size_t i = 0; i < attributes.size(); ++i) {
294  auto& typeName = attributes[i]->getTypeName();
295  if (!typeEnv.isType(typeName)) {
296  continue;
297  }
298 
299  auto argTypes = typeAnalysis.getTypes(arguments[i]);
300  auto& attributeType = typeEnv.getType(typeName);
301 
302  if (argTypes.isAll() || argTypes.empty()) {
303  continue; // This will be reported later.
304  }
305 
306  // We consider two cases: negated and not negated atoms.
307  // Negated atom have to agree in kind, non-negated atom need to follow source/sink rules.
308  if (negatedAtoms.count(&atom) == 0) {
309  // Attribute and argument type agree if, argument type is a subtype of declared type
310  // or is of the appropriate constant type or the (constant) record type.
311  bool validAttribute = all_of(argTypes, [&attributeType](const analysis::Type& type) {
312  if (isSubtypeOf(type, attributeType)) return true;
313  if (!isSubtypeOf(attributeType, type)) return false;
314  if (isA<ConstantType>(type)) return true;
315  return isA<analysis::RecordType>(type) && !isA<analysis::SubsetType>(type);
316  });
317 
318  if (!validAttribute && !Global::config().has("legacy")) {
319  auto primaryDiagnostic =
320  DiagnosticMessage("Atom's argument type is not a subtype of its declared type",
321  arguments[i]->getSrcLoc());
322 
323  auto declaredTypeInfo =
324  DiagnosticMessage(tfm::format("The argument's declared type is %s", typeName),
325  attributes[i]->getSrcLoc());
326 
327  report.addDiagnostic(Diagnostic(Diagnostic::Type::ERROR, std::move(primaryDiagnostic),
328  {std::move(declaredTypeInfo)}));
329  }
330  } else { // negation case.
331  // Declared attribute and deduced type agree if:
332  // They are the same type, or
333  // They are derived from the same constant type.
334  bool validAttribute = all_of(argTypes, [&](const analysis::Type& type) {
335  return type == attributeType || any_of(typeEnv.getConstantTypes(), [&](auto& constantType) {
336  return isSubtypeOf(attributeType, constantType) && isSubtypeOf(type, constantType);
337  });
338  });
339 
340  if (!validAttribute) {
341  auto primaryDiagnostic =
342  DiagnosticMessage("The kind of atom's argument doesn't match the declared type kind",
343  arguments[i]->getSrcLoc());
344  auto declaredTypeInfo =
345  DiagnosticMessage(tfm::format("The argument's declared type is %s", typeName),
346  attributes[i]->getSrcLoc());
347  report.addDiagnostic(Diagnostic(Diagnostic::Type::ERROR, std::move(primaryDiagnostic),
348  {std::move(declaredTypeInfo)}));
349  }
350  }
351  }
352 }

367 void TypeCheckerImpl::visitNumericConstant(const NumericConstant& constant) {
368  TypeSet types = typeAnalysis.getTypes(&constant);
369 
370  // No type could be assigned.
371  if (!constant.getType().has_value()) {
372  report.addError("Ambiguous constant (unable to deduce type)", constant.getSrcLoc());
373  return;
374  }
375 
376  switch (*constant.getType()) {
377  case NumericConstant::Type::Int:
378  if (!isOfKind(types, TypeAttribute::Signed)) {
379  report.addError("Number constant (type mismatch)", constant.getSrcLoc());
380  }
381  break;
382  case NumericConstant::Type::Uint:
383  if (!isOfKind(types, TypeAttribute::Unsigned)) {
384  report.addError("Unsigned constant (type mismatch)", constant.getSrcLoc());
385  }
386  break;
387  case NumericConstant::Type::Float:
388  if (!isOfKind(types, TypeAttribute::Float)) {
389  report.addError("Float constant (type mismatch)", constant.getSrcLoc());
390  }
391  break;
392  }

420 void TypeCheckerImpl::visitBranchInit(const BranchInit& adt) {
421  TypeSet types = typeAnalysis.getTypes(&adt);
422 
423  if (!isOfKind(types, TypeAttribute::ADT) || types.isAll() || types.size() != 1) {
424  report.addError("Ambiguous branch", adt.getSrcLoc());
425  return;
426  }
427 
428  // We know now that the set "types" is a singleton
429  auto& sumType = *as<analysis::AlgebraicDataType>(*types.begin());
430 
431  auto& argsDeclaredTypes = sumType.getBranchTypes(adt.getConstructor());
432  auto args = adt.getArguments();
433 
434  if (argsDeclaredTypes.size() != args.size()) {
435  // Invalid branchInit arity, handled by checkBranchInits.
436  return;
437  }
438 
439  for (size_t i = 0; i < args.size(); ++i) {
440  auto argTypes = typeAnalysis.getTypes(args[i]);
441  bool correctType = all_of(
442  argTypes, [&](const analysis::Type& t) { return isSubtypeOf(t, *argsDeclaredTypes[i]); });
443  if (!correctType) {
444  // TODO (darth_tytus): Give better error
445  report.addError("Branch argument's type doesn't match its declared type", args[i]->getSrcLoc());
446  }
447  }

484 void TypeCheckerImpl::visitUserDefinedFunctor(const UserDefinedFunctor& fun) {
485  // check type of result
486  const TypeSet& resultType = typeAnalysis.getTypes(&fun);
487 
488  TypeAttribute returnType;
489  try {
490  returnType = functorAnalysis.getReturnType(&fun);
491  } catch (...) {
492  report.addError("Undeclared user functor", fun.getSrcLoc());
493  return;
494  }
495 
496  if (!isOfKind(resultType, returnType)) {
497  switch (returnType) {
498  case TypeAttribute::Signed:
499  report.addError("Non-numeric use for numeric functor", fun.getSrcLoc());
500  break;
501  case TypeAttribute::Unsigned:
502  report.addError("Non-unsigned use for unsigned functor", fun.getSrcLoc());
503  break;
504  case TypeAttribute::Float:
505  report.addError("Non-float use for float functor", fun.getSrcLoc());
506  break;
507  case TypeAttribute::Symbol:
508  report.addError("Non-symbolic use for symbolic functor", fun.getSrcLoc());
509  break;
510  case TypeAttribute::Record: fatal("Invalid return type");
511  case TypeAttribute::ADT: fatal("Invalid return type");
512  }
513  }
514 
515  size_t i = 0;
516  for (auto arg : fun.getArguments()) {
517  TypeAttribute argType = functorAnalysis.getArgType(&fun, i);
518  if (!isOfKind(typeAnalysis.getTypes(arg), argType)) {
519  switch (argType) {
520  case TypeAttribute::Signed:
521  report.addError("Non-numeric argument for functor", arg->getSrcLoc());
522  break;
523  case TypeAttribute::Symbol:
524  report.addError("Non-symbolic argument for functor", arg->getSrcLoc());
525  break;
526  case TypeAttribute::Unsigned:
527  report.addError("Non-unsigned argument for functor", arg->getSrcLoc());
528  break;
529  case TypeAttribute::Float:
530  report.addError("Non-float argument for functor", arg->getSrcLoc());
531  break;
532  case TypeAttribute::Record: fatal("Invalid argument type");
533  case TypeAttribute::ADT: fatal("Invalid argument type");
534  }
535  }
536  ++i;
537  }

540 void TypeCheckerImpl::visitBinaryConstraint(const BinaryConstraint& constraint) {
541  auto op = constraint.getOperator();
542  auto left = constraint.getLHS();
543  auto right = constraint.getRHS();
544  auto opTypesAttrs = getBinaryConstraintTypes(op);
545 
546  auto leftTypes = typeAnalysis.getTypes(left);
547  auto rightTypes = typeAnalysis.getTypes(right);
548 
549  // Skip checks if either side is `Bottom` b/c it just adds noise.
550  // The unable-to-deduce-type checker will point out the issue.
551  if (leftTypes.empty() || rightTypes.empty() || leftTypes.isAll() || rightTypes.isAll()) return;
552 
553  assert((leftTypes.size() == 1) && (rightTypes.size() == 1));
554 
555  // Extract types from singleton sets.
556  auto& leftType = *typeAnalysis.getTypes(left).begin();
557  auto& rightType = *typeAnalysis.getTypes(right).begin();
558 
559  // give them a slightly nicer error
560  if (isOrderedBinaryConstraintOp(op) && !areEquivalentTypes(leftType, rightType)) {
561  report.addError("Cannot compare different types", constraint.getSrcLoc());
562  } else {
563  auto checkTyAttr = [&](Argument const& side) {
564  auto opMatchesType = any_of(opTypesAttrs,
565  [&](auto& typeAttr) { return isOfKind(typeAnalysis.getTypes(&side), typeAttr); });
566 
567  if (!opMatchesType) {
568  std::stringstream ss;
569  ss << "Constraint requires an operand of type "
570  << join(opTypesAttrs, " or ", [&](auto& out, auto& typeAttr) {
571  switch (typeAttr) {
572  case TypeAttribute::Signed: out << "`number`"; break;
573  case TypeAttribute::Symbol: out << "`symbol`"; break;
574  case TypeAttribute::Unsigned: out << "`unsigned`"; break;
575  case TypeAttribute::Float: out << "`float`"; break;
576  case TypeAttribute::Record: out << "a record"; break;
577  case TypeAttribute::ADT: out << "a sum"; break;
578  }
579  });
580  report.addError(ss.str(), side.getSrcLoc());
581  }
582  };
583 
584  checkTyAttr(*left);
585  checkTyAttr(*right);
586  }
587 }

36  *
37  * Example:
38  * .type A = B1 | B2 | ... | Bk

40  * A union type combines multiple types into a new super type.
41  * Each of the enumerated types become a sub-type of the new
42  * union type.
43  */

47  : Type(std::move(name), std::move(loc)), types(std::move(types)) {}

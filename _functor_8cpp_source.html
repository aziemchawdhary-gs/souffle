26 void FunctorAnalysis::run(const TranslationUnit& translationUnit) {
27  Program& program = translationUnit.getProgram();
28  const TypeAnalysis& typeAnalysis = *translationUnit.getAnalysis<TypeAnalysis>();
29 
30  visitDepthFirst(
31  program, [&](const FunctorDeclaration& fdecl) { udfDeclaration[fdecl.getName()] = &fdecl; });
32 
33  visitDepthFirst(program, [&](const IntrinsicFunctor& functor) {
34  // any valid candidate will do. pick the first.
35  try {
36  auto candidates = validOverloads(typeAnalysis, functor);
37  if (!candidates.empty()) {
38  functorInfo[functor.getFunction()] = &candidates.front().get();
39  }
40  } catch (...) {
41  // type analysis in validOverloads failed.
42  }
43  });
44 }

76 bool FunctorAnalysis::isMultiResult(const Functor& functor) {
77  if (isA<UserDefinedFunctor>(functor)) {
78  return false;
79  } else if (auto* intrinsic = as<IntrinsicFunctor>(functor)) {
80  auto op = intrinsic->getFunctionOp();

83  fatal("Missing functor type.");

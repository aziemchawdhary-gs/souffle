36 bool ADTtoRecordsTransformer::transform(TranslationUnit& tu) {
37  struct ADTsFuneral : public NodeMapper {
38  mutable bool changed{false};
39  TranslationUnit& tu;
40  const analysis::SumTypeBranchesAnalysis& sumTypesBranches =
41  *tu.getAnalysis<analysis::SumTypeBranchesAnalysis>();
42 
43  ADTsFuneral(TranslationUnit& tu) : tu(tu){};
44 
45  Own<Node> operator()(Own<Node> node) const override {
46  // Rewrite sub-expressions first
47  node->apply(*this);
48 
49  if (!isA<BranchInit>(node)) {
50  return node;
51  }
52 
53  changed = true;
54 
55  auto& adt = *as<BranchInit>(node);
56 
57  auto& type = sumTypesBranches.unsafeGetType(adt.getConstructor());
58 
59  auto& branches = type.getBranches();
60 
61  // Find branch ID.
62  analysis::AlgebraicDataType::Branch searchDummy{adt.getConstructor(), {}};
63  auto iterToBranch = std::lower_bound(branches.begin(), branches.end(), searchDummy,
64  [](const analysis::AlgebraicDataType::Branch& left,
65  const analysis::AlgebraicDataType::Branch& right) {
66  return left.name < right.name;
67  });
68 
69  // Branch id corresponds to the position in lexicographical ordering.
70  auto branchID = std::distance(std::begin(branches), iterToBranch);
71 
72  if (isADTEnum(type)) {
73  return mk<NumericConstant>(branchID);
74  } else {
75  // Collect branch arguments
76  VecOwn<Argument> branchArguments;
77  for (auto* arg : adt.getArguments()) {
78  branchArguments.emplace_back(arg->clone());
79  }
80 
81  // Branch is stored either as [branch_id, [arguments]]
82  // or [branch_id, argument] in case of a single argument.
83  auto branchArgs = [&]() -> Own<ast::Argument> {
84  if (branchArguments.size() != 1) {
85  return mk<Argument, RecordInit>(std::move(branchArguments));
86  } else {
87  return std::move(branchArguments.at(0));
88  }
89  }();
90 
91  // Arguments for the resulting record [branch_id, branch_args].
92  VecOwn<Argument> finalRecordArgs;
93 
94  finalRecordArgs.push_back(mk<Argument, NumericConstant>(branchID));
95  finalRecordArgs.push_back(std::move(branchArgs));
96 
97  return mk<RecordInit>(std::move(finalRecordArgs), adt.getSrcLoc());
98  }
99  }
100  };
101 
102  ADTsFuneral mapper(tu);
103  tu.getProgram().apply(mapper);
104  return mapper.changed;
105 }

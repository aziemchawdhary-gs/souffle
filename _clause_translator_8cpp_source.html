53 /** generate RAM code for a clause */
54 Own<ram::Statement> ClauseTranslator::translateClause(
55  const ast::Clause& clause, const ast::Clause& originalClause, const int version) {
56  if (auto reorderedClause = getReorderedClause(clause, version)) {
57  // translate reordered clause
58  return translateClause(*reorderedClause, originalClause, version);
59  }
60 
61  // get extract some details
62  const ast::Atom* head = clause.getHead();
63 
64  // handle facts
65  if (isFact(clause)) {
66  // translate arguments
67  VecOwn<ram::Expression> values;
68  for (auto& arg : head->getArguments()) {
69  values.push_back(translator.translateValue(arg, ValueIndex()));
70  }
71 
72  // create a fact statement
73  return mk<ram::Query>(mk<ram::Project>(translator.translateRelation(head), std::move(values)));
74  }
75 
76  // the rest should be rules
77  assert(isRule(clause));
78 
79  createValueIndex(clause);
80 
81  // -- create RAM statement --
82 
83  Own<ram::Operation> op = createOperation(clause);
84 
85  /* add equivalence constraints imposed by variable binding */
86  for (const auto& cur : valueIndex->getVariableReferences()) {
87  // the first appearance
88  const Location& first = *cur.second.begin();
89  // all other appearances
90  for (const Location& loc : cur.second) {
91  if (first != loc && !valueIndex->isGenerator(loc.identifier)) {
92  // FIXME: equiv' for float types (`FEQ`)
93  op = mk<ram::Filter>(
94  mk<ram::Constraint>(BinaryConstraintOp::EQ, translator.makeRamTupleElement(first),
95  translator.makeRamTupleElement(loc)),
96  std::move(op));
97  }
98  }
99  }
100 
101  /* add conditions caused by atoms, negations, and binary relations */
102  for (const auto& lit : clause.getBodyLiterals()) {
103  if (auto condition = translator.translateConstraint(lit, *valueIndex)) {
104  op = mk<ram::Filter>(std::move(condition), std::move(op));
105  }
106  }
107 
108  // add aggregator conditions
109  size_t curLevel = op_nesting.size() - 1;
110  for (auto it = op_nesting.rbegin(); it != op_nesting.rend(); ++it, --curLevel) {
111  const ast::Node* cur = *it;
112 
113  if (const auto* atom = dynamic_cast<const ast::Atom*>(cur)) {
114  // add constraints
115  size_t pos = 0;
116  for (auto arg : atom->getArguments()) {
117  if (auto* agg = dynamic_cast<ast::Aggregator*>(arg)) {
118  auto loc = valueIndex->getGeneratorLoc(*agg);
119  // FIXME: equiv' for float types (`FEQ`)
120  op = mk<ram::Filter>(
121  mk<ram::Constraint>(BinaryConstraintOp::EQ, mk<ram::TupleElement>(curLevel, pos),
122  translator.makeRamTupleElement(loc)),
123  std::move(op));
124  }
125  ++pos;
126  }
127  }
128  }
129 
130  // add generator levels
131  --level;
132  for (auto* cur : reverse(generators)) {
133  if (auto agg = dynamic_cast<const ast::Aggregator*>(cur)) {
134  // condition for aggregate and helper function to add terms
135  Own<ram::Condition> aggCond;
136  auto addAggCondition = [&](Own<ram::Condition> arg) {
137  aggCond = aggCond ? mk<ram::Conjunction>(std::move(aggCond), std::move(arg)) : std::move(arg);
138  };
139 
140  // translate constraints of sub-clause
141  for (auto&& lit : agg->getBodyLiterals()) {
142  if (auto newCondition = translator.translateConstraint(lit, *valueIndex)) {
143  addAggCondition(std::move(newCondition));
144  }
145  }
146 
147  // get the first predicate of the sub-clause
148  // NB: at most one atom is permitted in a sub-clause
149  const ast::Atom* atom = nullptr;
150  for (auto&& lit : agg->getBodyLiterals()) {
151  if (atom == nullptr) {
152  atom = dynamic_cast<const ast::Atom*>(lit);
153  } else {
154  assert(!isA<ast::Atom>(lit) && "Unsupported complex aggregation body encountered!");
155  }
156  }
157 
158  // translate arguments's of atom (if exists) to conditions
159  if (atom != nullptr) {
160  size_t pos = 0;
161  auto addAggEqCondition = [&](Own<ram::Expression> value) {
162  if (isUndefValue(value.get())) return;
163 
164  // FIXME: equiv' for float types (`FEQ`)
165  addAggCondition(mk<ram::Constraint>(
166  BinaryConstraintOp::EQ, mk<ram::TupleElement>(level, pos), std::move(value)));
167  };
168  for (auto* arg : atom->getArguments()) {
169  // variable bindings are issued differently since we don't want self
170  // referential variable bindings
171  if (auto* var = dynamic_cast<const ast::Variable*>(arg)) {
172  for (auto&& loc : valueIndex->getVariableReferences().find(var->getName())->second) {
173  if (level != loc.identifier || (int)pos != loc.element) {
174  addAggEqCondition(translator.makeRamTupleElement(loc));
175  break;
176  }
177  }
178  } else if (auto value = translator.translateValue(arg, *valueIndex)) {
179  addAggEqCondition(std::move(value));
180  }
181  ++pos;
182  }
183  }
184 
185  // translate aggregate expression
186  auto expr = translator.translateValue(agg->getTargetExpression(), *valueIndex);
187 
188  // add Ram-Aggregation layer
189  op = mk<ram::Aggregate>(std::move(op), agg->getOperator(), translator.translateRelation(atom),
190  expr ? std::move(expr) : mk<ram::UndefValue>(),
191  aggCond ? std::move(aggCond) : mk<ram::True>(), level);
192  } else if (const auto* func = dynamic_cast<const ast::IntrinsicFunctor*>(cur)) {
193  VecOwn<ram::Expression> args;
194  for (auto&& x : func->getArguments()) {
195  args.push_back(translator.translateValue(x, *valueIndex));
196  }
197 
198  auto func_op = [&]() -> ram::NestedIntrinsicOp {
199  switch (func->getFunctionOp().value()) {
200  case FunctorOp::RANGE: return ram::NestedIntrinsicOp::RANGE;
201  case FunctorOp::URANGE: return ram::NestedIntrinsicOp::URANGE;
202  case FunctorOp::FRANGE: return ram::NestedIntrinsicOp::FRANGE;
203 
204  default: fatal("missing case handler or bad code-gen");
205  }
206  };
207 
208  op = mk<ram::NestedIntrinsicOperator>(func_op(), std::move(args), std::move(op), level);
209  }
210 
211  --level;
212  }
213 
214  // build operation bottom-up
215  while (!op_nesting.empty()) {
216  // get next operator
217  const ast::Node* cur = op_nesting.back();
218  op_nesting.pop_back();
219 
220  // get current nesting level
221  auto level = op_nesting.size();
222 
223  if (const auto* atom = dynamic_cast<const ast::Atom*>(cur)) {
224  // add constraints
225  // TODO: do we wish to enable constraints by header functor? record inits do so...
226  op = filterByConstraints(level, atom->getArguments(), std::move(op), false);
227 
228  // check whether all arguments are unnamed variables
229  bool isAllArgsUnnamed = true;
230  for (auto* argument : atom->getArguments()) {
231  if (!isA<ast::UnnamedVariable>(argument)) {
232  isAllArgsUnnamed = false;
233  }
234  }
235 
236  // add check for emptiness for an atom
237  op = mk<ram::Filter>(
238  mk<ram::Negation>(mk<ram::EmptinessCheck>(translator.translateRelation(atom))),
239  std::move(op));
240 
241  // add a scan level
242  if (atom->getArity() != 0 && !isAllArgsUnnamed) {
243  if (head->getArity() == 0) {
244  op = mk<ram::Break>(
245  mk<ram::Negation>(mk<ram::EmptinessCheck>(translator.translateRelation(head))),
246  std::move(op));
247  }
248  if (Global::config().has("profile")) {
249  std::stringstream ss;
250  ss << head->getQualifiedName();
251  ss.str("");
252  ss << "@frequency-atom" << ';';
253  ss << originalClause.getHead()->getQualifiedName() << ';';
254  ss << version << ';';
255  ss << stringify(toString(clause)) << ';';
256  ss << stringify(toString(*atom)) << ';';
257  ss << stringify(toString(originalClause)) << ';';
258  ss << level << ';';
259  op = mk<ram::Scan>(translator.translateRelation(atom), level, std::move(op), ss.str());
260  } else {
261  op = mk<ram::Scan>(translator.translateRelation(atom), level, std::move(op));
262  }
263  }
264 
265  // TODO: support constants in nested records!
266  } else if (const auto* rec = dynamic_cast<const ast::RecordInit*>(cur)) {
267  // add constant constraints
268  op = filterByConstraints(level, rec->getArguments(), std::move(op));
269 
270  // add an unpack level
271  const Location& loc = valueIndex->getDefinitionPoint(*rec);
272  op = mk<ram::UnpackRecord>(
273  std::move(op), level, translator.makeRamTupleElement(loc), rec->getArguments().size());
274  } else {
275  fatal("Unsupported AST node for creation of scan-level!");
276  }
277  }
278 
279  /* generate the final RAM Insert statement */
280  Own<ram::Condition> cond = createCondition(originalClause);
281  if (cond != nullptr) {
282  return mk<ram::Query>(mk<ram::Filter>(std::move(cond), std::move(op)));
283  } else {
284  return mk<ram::Query>(std::move(op));
285  }
286 }

288 Own<ram::Operation> ClauseTranslator::createOperation(const ast::Clause& clause) {
289  const auto head = clause.getHead();
290 
291  VecOwn<ram::Expression> values;
292  for (ast::Argument* arg : head->getArguments()) {
293  values.push_back(translator.translateValue(arg, *valueIndex));
294  }
295 
296  Own<ram::Operation> project = mk<ram::Project>(translator.translateRelation(head), std::move(values));
297 
298  if (head->getArity() == 0) {
299  project = mk<ram::Filter>(
300  mk<ram::EmptinessCheck>(translator.translateRelation(head)), std::move(project));
301  }
302 
303  // build up insertion call
304  return project; // start with innermost
305 }

307 Own<ram::Condition> ClauseTranslator::createCondition(const ast::Clause& originalClause) {
308  const auto head = originalClause.getHead();
309 
310  // add stopping criteria for nullary relations
311  // (if it contains already the null tuple, don't re-compute)
312  if (head->getArity() == 0) {
313  return mk<ram::EmptinessCheck>(translator.translateRelation(head));
314  }
315  return nullptr;

318 Own<ram::Operation> ClauseTranslator::filterByConstraints(size_t const level,
319  const std::vector<ast::Argument*>& args, Own<ram::Operation> op, bool constrainByFunctors) {
320  size_t pos = 0;
321 
322  auto mkFilter = [&](bool isFloatArg, Own<ram::Expression> rhs) {
323  return mk<ram::Filter>(
324  mk<ram::Constraint>(isFloatArg ? BinaryConstraintOp::FEQ : BinaryConstraintOp::EQ,
325  mk<ram::TupleElement>(level, pos), std::move(rhs)),
326  std::move(op));
327  };
328 
329  for (auto* a : args) {
330  if (auto* c = dynamic_cast<const ast::Constant*>(a)) {
331  auto* const c_num = dynamic_cast<const ast::NumericConstant*>(c);
332  assert((!c_num || c_num->getType()) && "numeric constant wasn't bound to a type");
333  op = mkFilter(c_num && *c_num->getType() == ast::NumericConstant::Type::Float,
334  translator.translateConstant(*c));
335  } else if (auto* func = dynamic_cast<const ast::Functor*>(a)) {
336  if (constrainByFunctors) {
337  TypeAttribute returnType = translator.getFunctorAnalysis()->getReturnType(func);
338  op = mkFilter(
339  returnType == TypeAttribute::Float, translator.translateValue(func, *valueIndex));
340  }
341  }
342 
343  ++pos;
344  }
345 
346  return op;
347 }

349 Own<ast::Clause> ClauseTranslator::getReorderedClause(const ast::Clause& clause, const int version) const {
350  const auto plan = clause.getExecutionPlan();
351 
352  // check whether there is an imposed order constraint
353  if (plan == nullptr) {
354  // no plan, so reorder it according to the internal heuristic
355  if (auto* reorderedClause = ast::transform::ReorderLiteralsTransformer::reorderClauseWithSips(
356  *translator.getSipsMetric(), &clause)) {
357  return Own<ast::Clause>(reorderedClause);
358  }
359  return nullptr;
360  }
361  auto orders = plan->getOrders();
362  if (orders.find(version) == orders.end()) {
363  return nullptr;
364  }
365 
366  // get the imposed order
367  const auto& order = orders[version];
368 
369  // create a copy and fix order
370  Own<ast::Clause> reorderedClause(clause.clone());
371 
372  // Change order to start at zero
373  std::vector<unsigned int> newOrder(order->getOrder().size());
374  std::transform(order->getOrder().begin(), order->getOrder().end(), newOrder.begin(),
375  [](unsigned int i) -> unsigned int { return i - 1; });
376 
377  // re-order atoms
378  reorderedClause.reset(reorderAtoms(reorderedClause.get(), newOrder));
379 
380  // clear other order and fix plan
381  reorderedClause->clearExecutionPlan();
382 
383  return reorderedClause;
384 }

386 void ClauseTranslator::indexValues(const ast::Node* curNode, const std::vector<ast::Argument*>& curNodeArgs,
387  std::map<const ast::Node*, int>& nodeLevel, const ram::Relation* relation) {
388  for (size_t pos = 0; pos < curNodeArgs.size(); ++pos) {
389  // get argument
390  auto& arg = curNodeArgs[pos];
391 
392  // check for variable references
393  if (auto var = dynamic_cast<const ast::Variable*>(arg)) {
394  if (pos < relation->getArity()) {
395  valueIndex->addVarReference(*var, nodeLevel[curNode], pos, relation->getName());
396  } else {
397  valueIndex->addVarReference(*var, nodeLevel[curNode], pos);
398  }
399  }
400 
401  // check for nested records
402  if (auto rec = dynamic_cast<const ast::RecordInit*>(arg)) {
403  // introduce new nesting level for unpack
404  op_nesting.push_back(rec);
405  nodeLevel[rec] = level++;
406 
407  // register location of record
408  valueIndex->setRecordDefinition(*rec, nodeLevel[curNode], pos);
409 
410  // resolve nested components
411  indexValues(rec, rec->getArguments(), nodeLevel, relation);
412  }
413  }

416 /** index values in rule */
417 void ClauseTranslator::createValueIndex(const ast::Clause& clause) {
418  for (const auto* atom : ast::getBodyLiterals<ast::Atom>(clause)) {
419  // map from each list of arguments to its nesting level
420  std::map<const ast::Node*, int> nodeLevel;
421 
422  // give the atom the current level
423  nodeLevel[atom] = level++;
424  op_nesting.push_back(atom);
425 
426  // index each value in the atom
427  indexValues(atom, atom->getArguments(), nodeLevel,
428  translator.lookupRelation(translator.translateRelation(atom)));
429  }
430 
431  // add aggregation functions
432  visitDepthFirstPostOrder(clause, [&](const ast::Argument& arg) {
433  // returns the write-location for this generator (or none if an equiv arg was already seen)
434  auto addGenerator = [&]() -> std::optional<int> {
435  // The by-value compare means that we're effectively doing CSE for any
436  // generator args during code-gen. This is a weird place to do this.
437  if (dynamic_cast<const ast::Aggregator*>(&arg) != nullptr &&
438  any_of(generators, [&](auto* x) { return *x == arg; }))
439  return {};
440  generators.push_back(&arg);
441 
442  int aggLoc = level++;
443  valueIndex->setGeneratorLoc(arg, Location({aggLoc, 0}));
444  return aggLoc;
445  };
446 
447  if (auto agg = dynamic_cast<const ast::Aggregator*>(&arg)) {
448  if (auto aggLoc = addGenerator()) {
449  // bind aggregator variables to locations
450  const ast::Atom* atom = nullptr;
451  for (auto lit : agg->getBodyLiterals()) {
452  if (atom == nullptr) {
453  atom = dynamic_cast<const ast::Atom*>(lit);
454  } else {
455  break;
456  }
457  }
458  if (atom != nullptr) {
459  size_t pos = 0;
460  for (auto* arg : atom->getArguments()) {
461  if (const auto* var = dynamic_cast<const ast::Variable*>(arg)) {
462  valueIndex->addVarReference(
463  *var, *aggLoc, (int)pos, translator.translateRelation(atom));
464  }
465  ++pos;
466  }
467  }
468  }
469  }
470 
471  auto* func = as<ast::IntrinsicFunctor>(arg);
472  if (func && ast::analysis::FunctorAnalysis::isMultiResult(*func)) {
473  addGenerator();
474  }
475  });
476 
477  // add multi-result functor introductions
478  visitDepthFirst(clause, [&](const ast::BinaryConstraint& bc) {
479  if (bc.getOperator() != BinaryConstraintOp::EQ && bc.getOperator() != BinaryConstraintOp::FEQ) return;
480  const auto* lhs = dynamic_cast<const ast::Variable*>(bc.getLHS());
481  const auto* rhs = dynamic_cast<const ast::IntrinsicFunctor*>(bc.getRHS());
482  if (lhs == nullptr || rhs == nullptr) return;
483  if (!ast::analysis::FunctorAnalysis::isMultiResult(*rhs)) return;
484  valueIndex->addVarReference(*lhs, valueIndex->getGeneratorLoc(*rhs));
485  });
486 }

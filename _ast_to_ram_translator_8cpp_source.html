341  /** for provenance negation */
342  Own<ram::Condition> visitProvenanceNegation(const ast::ProvenanceNegation& neg) override {
343  const auto* atom = neg.getAtom();
344  size_t auxiliaryArity = translator.getEvaluationArity(atom);
345  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
346  size_t arity = atom->getArity() - auxiliaryArity;
347  VecOwn<ram::Expression> values;
348 
349  auto args = atom->getArguments();
350  for (size_t i = 0; i < arity; i++) {
351  values.push_back(translator.translateValue(args[i], index));
352  }
353  // we don't care about the provenance columns when doing the existence check
354  if (Global::config().has("provenance")) {
355  // undefined value for rule number
356  values.push_back(mk<ram::UndefValue>());
357  // add the height annotation for provenanceNotExists
358  for (size_t height = 1; height < auxiliaryArity; height++) {
359  values.push_back(translator.translateValue(args[arity + height], index));
360  }
361  }
362  return mk<ram::Negation>(
363  mk<ram::ProvenanceExistenceCheck>(translator.translateRelation(atom), std::move(values)));
364  }
365 
366  /** for negations */
367  Own<ram::Condition> visitNegation(const ast::Negation& neg) override {
368  const auto* atom = neg.getAtom();
369  size_t auxiliaryArity = translator.getEvaluationArity(atom);
370  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
371  size_t arity = atom->getArity() - auxiliaryArity;
372 
373  if (arity == 0) {
374  // for a nullary, negation is a simple emptiness check
375  return mk<ram::EmptinessCheck>(translator.translateRelation(atom));
376  }
377 
378  // else, we construct the atom and create a negation
379  VecOwn<ram::Expression> values;
380  auto args = atom->getArguments();
381  for (size_t i = 0; i < arity; i++) {
382  values.push_back(translator.translateValue(args[i], index));
383  }
384  for (size_t i = 0; i < auxiliaryArity; i++) {
385  values.push_back(mk<ram::UndefValue>());
386  }
387  return mk<ram::Negation>(
388  mk<ram::ExistenceCheck>(translator.translateRelation(atom), std::move(values)));
389  }
390  };
391  return ConstraintTranslator(*this, index)(*lit);
392 }

427 /** generate RAM code for a non-recursive relation */
428 Own<ram::Statement> AstToRamTranslator::translateNonRecursiveRelation(
429  const ast::Relation& rel, const ast::analysis::RecursiveClausesAnalysis* recursiveClauses) {
430  /* start with an empty sequence */
431  VecOwn<ram::Statement> res;
432 
433  std::string relName = translateRelation(&rel);
434 
435  /* iterate over all clauses that belong to the relation */
436  for (ast::Clause* clause : getClauses(*program, rel)) {
437  // skip recursive rules
438  if (recursiveClauses->recursive(clause)) {
439  continue;
440  }
441 
442  // translate clause
443  Own<ram::Statement> rule = ClauseTranslator(*this).translateClause(*clause, *clause);
444 
445  // add logging
446  if (Global::config().has("profile")) {
447  const std::string& relationName = toString(rel.getQualifiedName());
448  const SrcLocation& srcLocation = clause->getSrcLoc();
449  const std::string clauseText = stringify(toString(*clause));
450  const std::string logTimerStatement =
451  LogStatement::tNonrecursiveRule(relationName, srcLocation, clauseText);
452  const std::string logSizeStatement =
453  LogStatement::nNonrecursiveRule(relationName, srcLocation, clauseText);
454  rule = mk<ram::LogRelationTimer>(std::move(rule), logTimerStatement, relName);
455  }
456 
457  // add debug info
458  std::ostringstream ds;
459  ds << toString(*clause) << "\nin file ";
460  ds << clause->getSrcLoc();
461  rule = mk<ram::DebugInfo>(std::move(rule), ds.str());
462 
463  // add rule to result
464  appendStmt(res, std::move(rule));
465  }
466 
467  // add logging for entire relation
468  if (Global::config().has("profile")) {
469  const std::string& relationName = toString(rel.getQualifiedName());
470  const SrcLocation& srcLocation = rel.getSrcLoc();
471  const std::string logSizeStatement = LogStatement::nNonrecursiveRelation(relationName, srcLocation);
472 
473  // add timer if we did any work
474  if (!res.empty()) {
475  const std::string logTimerStatement =
476  LogStatement::tNonrecursiveRelation(relationName, srcLocation);
477  auto newStmt =
478  mk<ram::LogRelationTimer>(mk<ram::Sequence>(std::move(res)), logTimerStatement, relName);
479  res.clear();
480  appendStmt(res, std::move(newStmt));
481  } else {
482  // add table size printer
483  appendStmt(res, mk<ram::LogSize>(relName, logSizeStatement));
484  }
485  }
486 
487  // done
488  return mk<ram::Sequence>(std::move(res));
489 }

529 /** generate RAM code for recursive relations in a strongly-connected component */
530 Own<ram::Statement> AstToRamTranslator::translateRecursiveRelation(const std::set<const ast::Relation*>& scc,
531  const ast::analysis::RecursiveClausesAnalysis* recursiveClauses) {
532  // initialize sections
533  VecOwn<ram::Statement> preamble;
534  VecOwn<ram::Statement> updateTable;
535  VecOwn<ram::Statement> postamble;
536 
537  auto genMerge = [&](const ast::Relation* rel, const std::string& destRel,
538  const std::string& srcRel) -> Own<ram::Statement> {
539  VecOwn<ram::Expression> values;
540  if (rel->getArity() == 0) {
541  return mk<ram::Query>(mk<ram::Filter>(mk<ram::Negation>(mk<ram::EmptinessCheck>(srcRel)),
542  mk<ram::Project>(destRel, std::move(values))));
543  }
544  for (std::size_t i = 0; i < rel->getArity(); i++) {
545  values.push_back(mk<ram::TupleElement>(0, i));
546  }
547  auto stmt = mk<ram::Query>(mk<ram::Scan>(srcRel, 0, mk<ram::Project>(destRel, std::move(values))));
548  if (rel->getRepresentation() == RelationRepresentation::EQREL) {
549  return mk<ram::Sequence>(mk<ram::Extend>(destRel, srcRel), std::move(stmt));
550  }
551  return stmt;
552  };
553 
554  // --- create preamble ---
555 
556  /* Compute non-recursive clauses for relations in scc and push
557  the results in their delta tables. */
558  for (const ast::Relation* rel : scc) {
559  /* create update statements for fixpoint (even iteration) */
560  Own<ram::Statement> updateRelTable =
561  mk<ram::Sequence>(genMerge(rel, translateRelation(rel), translateNewRelation(rel)),
562  mk<ram::Swap>(translateDeltaRelation(rel), translateNewRelation(rel)),
563  mk<ram::Clear>(translateNewRelation(rel)));
564 
565  /* measure update time for each relation */
566  if (Global::config().has("profile")) {
567  updateRelTable = mk<ram::LogRelationTimer>(std::move(updateRelTable),
568  LogStatement::cRecursiveRelation(toString(rel->getQualifiedName()), rel->getSrcLoc()),
569  translateNewRelation(rel));
570  }
571 
572  /* drop temporary tables after recursion */
573  appendStmt(postamble, mk<ram::Clear>(translateDeltaRelation(rel)));
574  appendStmt(postamble, mk<ram::Clear>(translateNewRelation(rel)));
575 
576  /* Generate code for non-recursive part of relation */
577  /* Generate merge operation for temp tables */
578  appendStmt(preamble, translateNonRecursiveRelation(*rel, recursiveClauses));
579  appendStmt(preamble, genMerge(rel, translateDeltaRelation(rel), translateRelation(rel)));
580 
581  /* Add update operations of relations to parallel statements */
582  appendStmt(updateTable, std::move(updateRelTable));
583  }
584 
585  // --- build main loop ---
586 
587  VecOwn<ram::Statement> loopSeq;
588 
589  // create a utility to check SCC membership
590  auto isInSameSCC = [&](const ast::Relation* rel) {
591  return std::find(scc.begin(), scc.end(), rel) != scc.end();
592  };
593 
594  /* Compute temp for the current tables */
595  for (const ast::Relation* rel : scc) {
596  VecOwn<ram::Statement> loopRelSeq;
597 
598  /* Find clauses for relation rel */

631  // atoms)
632  nameUnnamedVariables(r1.get());
633

635  for (size_t k = j + 1; k < atoms.size(); k++) {
636  if (isInSameSCC(getAtomRelation(atoms[k], program))) {
637  auto cur = souffle::clone(ast::getBodyLiterals<ast::Atom>(*r1)[k]);
638  cur->setQualifiedName(translateDeltaRelation(getAtomRelation(atoms[k], program)));
639  r1->addToBody(mk<ast::Negation>(std::move(cur)));
640  }
641  }

644 
645  /* add logging */

647  const std::string& relationName = toString(rel->getQualifiedName());
648  const SrcLocation& srcLocation = cl->getSrcLoc();
649  const std::string clauseText = stringify(toString(*cl));

651  LogStatement::tRecursiveRule(relationName, version, srcLocation, clauseText);
652  const std::string logSizeStatement =
653  LogStatement::nRecursiveRule(relationName, version, srcLocation, clauseText);
654  rule = mk<ram::LogRelationTimer>(
655  std::move(rule), logTimerStatement, translateNewRelation(rel));
656  }

660  ds << toString(*cl) << "\nin file ";
661  ds << cl->getSrcLoc();

665  appendStmt(loopRelSeq, std::move(rule));
666 
667  // increment version counter
668  version++;
669  }

673  int maxVersion = -1;

675  maxVersion = std::max(cur.first, maxVersion);
676  }

678  }
679  }

682  if (loopRelSeq.size() == 0) {
683  continue;
684  }

686  // label all versions
687  if (Global::config().has("profile")) {
688  const std::string& relationName = toString(rel->getQualifiedName());
689  const SrcLocation& srcLocation = rel->getSrcLoc();
690  const std::string logTimerStatement = LogStatement::tRecursiveRelation(relationName, srcLocation);
691  const std::string logSizeStatement = LogStatement::nRecursiveRelation(relationName, srcLocation);
692  auto newStmt = mk<ram::LogRelationTimer>(
693  mk<ram::Sequence>(std::move(loopRelSeq)), logTimerStatement, translateNewRelation(rel));
694  loopRelSeq.clear();
695  appendStmt(loopRelSeq, std::move(newStmt));
696  }
697 
698  /* add rule computations of a relation to parallel statement */
699  appendStmt(loopSeq, mk<ram::Sequence>(std::move(loopRelSeq)));
700  }
701  auto loop = mk<ram::Parallel>(std::move(loopSeq));
702 
703  /* construct exit conditions for odd and even iteration */
704  auto addCondition = [](Own<ram::Condition>& cond, Own<ram::Condition> clause) {
705  cond = ((cond) ? mk<ram::Conjunction>(std::move(cond), std::move(clause)) : std::move(clause));
706  };
707 
708  Own<ram::Condition> exitCond;
709  VecOwn<ram::Statement> exitStmts;
710  for (const ast::Relation* rel : scc) {
711  addCondition(exitCond, mk<ram::EmptinessCheck>(translateNewRelation(rel)));
712  if (ioType->isLimitSize(rel)) {
713  Own<ram::Condition> limit =
714  mk<ram::Constraint>(BinaryConstraintOp::GE, mk<ram::RelationSize>(translateRelation(rel)),
715  mk<ram::SignedConstant>(ioType->getLimitSize(rel)));
716  appendStmt(exitStmts, mk<ram::Exit>(std::move(limit)));
717  }
718  }
719 
720  /* construct fixpoint loop */
721  VecOwn<ram::Statement> res;
722  if (preamble.size() > 0) {
723  appendStmt(res, mk<ram::Sequence>(std::move(preamble)));
724  }
725  if (!loop->getStatements().empty() && exitCond && updateTable.size() > 0) {
726  appendStmt(res,
727  mk<ram::Loop>(mk<ram::Sequence>(std::move(loop), mk<ram::Exit>(std::move(exitCond)),
728  mk<ram::Sequence>(std::move(exitStmts)), mk<ram::Sequence>(std::move(updateTable)))));
729  }
730  if (postamble.size() > 0) {
731  appendStmt(res, mk<ram::Sequence>(std::move(postamble)));
732  }
733  if (res.size() > 0) {
734  return mk<ram::Sequence>(std::move(res));
735  }

747  if (!isA<ast::Constraint>(bodyLit)) {
748  intermediateClause->addToBody(souffle::clone(bodyLit));
749  }
750  }

752  // now add all constraints
753  for (auto bodyLit : ast::getBodyLiterals<ast::Constraint>(clause)) {

755  }

789  auto arity = atom->getArity();
790  auto atomArgs = atom->getArguments();

792  intermediateClause->addToBody(mk<ast::BinaryConstraint>(BinaryConstraintOp::LT,
793  souffle::clone(atomArgs[arity - 1]), mk<ast::SubroutineArgument>(levelIndex)));
794  }
795  }

1133  (!isRecursive) ? translateNonRecursiveRelation(
1134  *((const ast::Relation*)*allInterns.begin()), recursiveClauses)
1135  : translateRecursiveRelation(allInterns, recursiveClauses);

1139  for (const auto& relation : internOuts) {
